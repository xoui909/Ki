//회원

CREATE TABLE USR(
	U_IDX 	  NUMBER(6,0) CONSTRAINT USR_PK PRIMARY KEY,
	U_ID      VARCHAR2(50),
	U_PW	  VARCHAR2(20),
	U_NAME    VARCHAR2(50),
        U_IMAGE   VARCHAR2(1000),
	U_BIRTH    VARCHAR2(50),
	U_GENDER    VARCHAR2(10),
	U_ADDR    VARCHAR2(100),
	U_TEL	  VARCHAR2(20),
	U_EMAIL   VARCHAR2(30),
	U_STATUS  VARCHAR2(1)
     );

//기(글)
CREATE TABLE KI(
	K_IDX 	  NUMBER(6,0) CONSTRAINT KI_PK PRIMARY KEY,
	U_IDX      NUMBER(6,0),
	K_KIND	  VARCHAR2(50),
	K_CONTENT    VARCHAR2(200),
	K_IMAGE     VARCHAR2(1000),
	K_MAX    VARCHAR2(100),
        K_HIT	  VARCHAR2(50),
	K_REGDATE DATE,
        CONSTRAINT KI_FK FOREIGN KEY (U_IDX) REFERENCES USR(U_IDX)
     );

//기받는사람
CREATE TABLE KI_EE(
	KI_IDX 	  NUMBER(6,0) CONSTRAINT KI_EE_PK PRIMARY KEY,
	KI_U_IDX      NUMBER(6,0),
	KI_REGDATE DATE
     );


//댓글
CREATE TABLE KI_RE(
	R_IDX 	  NUMBER(6,0) CONSTRAINT KI_RE_PK PRIMARY KEY,
	K_IDX      NUMBER(6,0),
	R_U_IDX	  VARCHAR2(50),
	R_CONTENT  VARCHAR2(2000),
	R_REGDATE DATE,
        CONSTRAINT KI_RE_FK FOREIGN KEY (K_IDX) REFERENCES KI(K_IDX)
     );

//소원
CREATE TABLE WISH(
	W_IDX 	  NUMBER(6,0) CONSTRAINT WISH_PK PRIMARY KEY,
	U_IDX      NUMBER(6,0),
	W_STATUS   VARCHAR2(1),
	W_CUR_KI    VARCHAR2(100),
	W_MAX_KI     VARCHAR2(100),
	W_CONTENT    VARCHAR2(1000),
	W_REGDATE DATE,
        CONSTRAINT WISH_FK FOREIGN KEY (U_IDX) REFERENCES USR(U_IDX)
     );


//친구
CREATE TABLE FRIEND(
	F_U_IDX 	  NUMBER(6,0),
	F_IDX      NUMBER(6,0)
     );

//시퀀스
create sequence usr_seq
	increment by 1
	start with 1;

create sequence ki_seq
	increment by 1
	start with 1;

create sequence ki_ee_seq
	increment by 1
	start with 1;

create sequence ki_re_seq
	increment by 1
	start with 1;

create sequence wish_seq
	increment by 1
	start with 1;
